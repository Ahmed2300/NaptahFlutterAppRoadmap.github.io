ğŸ“‹ Implementation

import { db } from '../lib/firebase';
import { ref, push, set, onValue, query, orderByChild } from 'firebase/database';

export function sendMessage(chatId: string, message: any) {
  const messagesRef = ref(db, `chats/${chatId}/messages`);
  const newMessageRef = push(messagesRef);
  return set(newMessageRef, {
    ...message,
    timestamp: Date.now()
  });
}

export function subscribeToMessages(chatId: string, callback: Function) {
  const messagesRef = query(
    ref(db, `chats/${chatId}/messages`),
    orderByChild('timestamp')
  );
  return onValue(messagesRef, (snapshot) => {
    const messages = [];
    snapshot.forEach((child) => {
      messages.push({ id: child.key, ...child.val() });
    });
    callback(messages);
  });
}
