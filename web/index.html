    import { GoogleGenerativeAI } from '@google/generative-ai';

const genAI = new GoogleGenerativeAI(import.meta.env.VITE_GEMINI_API_KEY);

export async function analyzePlantImage(imageData: string) {
  const model = genAI.getGenerativeModel({ model: 'gemini-1.5-flash' });
  
  const prompt = `Analyze this plant image and provide:
1. Plant identification
2. Disease detection (if any)
3. Health status
4. Treatment recommendations
5. Care instructions

Return in JSON format.`;

  const imagePart = {
    inlineData: {
      data: imageData.split(',')[1],
      mimeType: 'image/jpeg'
    }
  };

  const result = await model.generateContent([prompt, imagePart]);
  const response = result.response.text();
  
  return JSON.parse(response);
}

💬 Real-time Chat Service Guide

Assignee: Khaled | Estimated Time: 8-10 hours

🎯 Goal

Build real-time chat system for expert consultations using Firebase Realtime Database.

📋 Implementation

import { db } from '../lib/firebase';
import { ref, push, set, onValue, query, orderByChild } from 'firebase/database';

export function sendMessage(chatId: string, message: any) {
  const messagesRef = ref(db, `chats/${chatId}/messages`);
  const newMessageRef = push(messagesRef);
  return set(newMessageRef, {
    ...message,
    timestamp: Date.now()
  });
}

export function subscribeToMessages(chatId: string, callback: Function) {
  const messagesRef = query(
    ref(db, `chats/${chatId}/messages`),
    orderByChild('timestamp')
  );
  return onValue(messagesRef, (snapshot) => {
    const messages = [];
    snapshot.forEach((child) => {
      messages.push({ id: child.key, ...child.val() });
    });
    callback(messages);
  });
}
